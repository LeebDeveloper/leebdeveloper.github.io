Что хорошего в Erlang?

Иммутабельность /
single assignment

  • Присваивание Связывание
  • Элегантный pattern-matching
  • Можно написать NIF

Erlang


in_list(X, [X | _]) -> true;
in_list(X, [_ | L]) -> in_list(X, L);
in_list(_, [])      -> false.

...

case in_list(3, [1, 3, 5, 7]) of
	true  -> ...;
	false -> ...
end.
